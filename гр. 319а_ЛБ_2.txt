    Х – змінна, дійсний тип.
    Алгоритм:
    1) Виведення запрошення до вводу;

    Лістинг коду вирішення задачі 1

    #include<iostream>
    #include<cmath>
    using namespace std;
    int main()
    {
    // y = (tab. N29)
    // declarations
    double x, num, denom, y;
    // input
    cout << " Real argument x = ";
    cin >> x;
    // calc
    num = pow(abs((x*x*x)/4-pow(sin(x*x*x),2)*2),1/3)*log(5)*log(abs(x)) / log(5); // numerator
    denom = pow(2,x)*sqrt(abs(3*x+2.5)*abs(tan(x))); // denominator
    y = num/denom;
    // output
    cout << " Function y = " << y << endl;
    return 0;
    }

    Завдання 3. Вирішення задачі Boolean 10
    Вхідні дані (ім’я, опис, тип, обмеження):
    A - ціле число, дійсний тип;
    B - ціле число, дійсний тип.
    Вихідні дані (ім’я, опис, тип):
    Рівне число А, дійсний тип;
    Непарне число В, дійсний тип.
    Алгоритм:
    1) Виведення запрошення до вводу;
    2) Введення тризначного числа;
    3) Розрахунок результату «Рівне одне з чисел A i B непарне»;
    4) Виведення «Рівне одне з чисел A i B непарне» .
    

    Лістинг коду вирішення задачі Boolean 10:

    #include <iostream>
    #include <cmath>
    using namespace std;
    int main()
    {
    // Дано два цілих числа: A, B.
    // Перевірити істинність висловлювання: «Рівне одне з чисел A і B непарне».
    // declaratios
    int A, B;
    // input
    cout << " A ";
    cin >> A ;
    cout << " B ";
    cin >> B;
    // calc
    bool is_pos = ((A % 2 != 0 && B % 2 == 0) || (B % 2 != 0 && A % 2 == 0));
    // output
    cout << " One of the numbers is even = " << is_pos << endl;
    return 0;
    }
 
    Виконання завдання 3.
    A 5
    B 10
    One of the numbers is even = 1
